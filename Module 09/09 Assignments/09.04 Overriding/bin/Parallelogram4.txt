/**
 * This class defines a Parallelogram object
 *
 * @Tristan Darnell
 * @12/25/23
 */
public class Parallelogram4 extends Rectangle4{
    public Parallelogram4(int l, int w)
    {
        super(l, w);
    }

    public String toString()
    {
        return "Parallelogram - " + getLength() + " X " + getWidth();
    }
    
}
